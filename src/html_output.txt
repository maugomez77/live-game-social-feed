    #feed { background: white; border: 1px solid #ddd; border-radius: 8px; padding: 16px; height: 400px; overflow-y: auto; margin-bottom: 20px; }

    .comment .username { font-weight: bold; color: #2c3e50; }
    .comment .text { margin: 4px 0; }
    .comment .timestamp { font-size: 0.8em; color: #999; }
    #form { display: flex; gap: 8px; }

<body>
  <h1>🏟️ Live Game Commentary</h1>
  <div id="status" class="status">Connecting...</div>
  <div id="feed"></div>
  <div id="form">
    <input type="text" id="usernameInput" placeholder="Username" />
    <input type="text" id="textInput" placeholder="Write a comment..." />
    <button id="sendBtn">Send</button>
  </div>

  <script>
    const feed = document.getElementById('feed');
    const status = document.getElementById('status');
    const usernameInput = document.getElementById('usernameInput');
    const textInput = document.getElementById('textInput');
    const sendBtn = document.getElementById('sendBtn');

    const ws = new WebSocket('ws://localhost:4000');

    ws.onopen = () => {
      status.textContent = '✅ Connected';
      status.style.color = '#27ae60';
    };

    ws.onclose = () => {
      status.textContent = '❌ Disconnected. Refresh to reconnect.';
      status.style.color = '#e74c3c';
    };

    ws.onerror = (err) => {
      status.textContent = '⚠️ Connection error';
      status.style.color = '#e67e22';
    };

    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      if (data.type === 'history') {
        feed.innerHTML = '';
        data.comments.forEach(comment => appendComment(comment));
        feed.scrollTop = feed.scrollHeight;
      } else if (data.type === 'new_comment') {
        appendComment(data.comment);
        feed.scrollTop = feed.scrollHeight;
      } else if (data.type === 'error') {
        alert('Server error: ' + data.message);
      }
    };

    function appendComment(comment) {
      const div = document.createElement('div');
      div.className = 'comment';
      div.innerHTML = `
        <div class="username">${escapeHtml(comment.username)}</div>
        <div class="text">${escapeHtml(comment.text)}</div>
        <div class="timestamp">${new Date(comment.timestamp).toLocaleTimeString()}</div>
      `;
      feed.appendChild(div);
    }

    function escapeHtml(str) {
      const div = document.createElement('div');
      div.textContent = str;
      return div.innerHTML;
    }

    sendBtn.addEventListener('click', () => {
      const username = usernameInput.value.trim() || 'Anonymous';
      const text = textInput.value.trim();
      if (!text) return;

      ws.send(JSON.stringify({
        type: 'new_comment',
        username: username,
        text: text
      }));

      textInput.value = '';
    });

    textInput.addEventListener('keypress', (e) => {
      if (e.key === 'Enter') sendBtn.click();
    });
  </script>
</body>
